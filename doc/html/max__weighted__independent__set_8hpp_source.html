191  //first dimension

193  if (memory_it_d1 != memory.end()) {

195  auto memory_it_d2 = memory_it_d1->second.find(s);
196  if (memory_it_d2 != memory_it_d1->second.end()) return;

200  std::list<Graph_vertex> w;
201  std::list<Decomposition_vertex> neighbours = get_neighbours(d, t, parent);
202 
203  //leaf node
204  if (neighbours.size() == 0) {

206  return;
207  }
208  //join node

213  }

215  memory[t][s] = make_pair(res_weight, w);
216  return;
217  }

221  //creating s \ diff

223  if (s.find(diff) == s.end()) s_new = s;
224  else {

226  if (curr != diff) s_new.insert(curr);
227  }
228  res_weight = weights[diff];
229  }

231  res_weight += memory[neighbours.front()][s_new].first;
232  memory[t][s] = make_pair(res_weight, w);
233  return;
234  }
235  //forget node
236  else if (neighbours.size() == 1 && (bags[t].size() + 1 == bags[neighbours.front()].size())) {
237  Graph_vertex diff = get_diff(bags[neighbours.front()], bags[t]);
238  //creating s+diff
239  std::set<Graph_vertex> s_new = s;
240  s_new.insert(diff);
241  //check if s+diff is independent - if not, we can return
242  if (is_adjacent(g, s, diff)) {
243  calculate_weight(g, d, bags, weights, neighbours.front(), t, s, memory);
244  memory[t][s] = make_pair(memory[neighbours.front()][s].first, w);
245  }
246  else {
247  calculate_weight(g, d, bags, weights, neighbours.front(), t, s, memory);
248  calculate_weight(g, d, bags, weights, neighbours.front(), t, s_new, memory);
249 
250  if (memory[neighbours.front()][s_new].first > memory[neighbours.front()][s].first) {
251  w.push_back(diff);
252  res_weight = memory[neighbours.front()][s_new].first;
253  }
254  else res_weight = memory[neighbours.front()][s].first;
255  memory[t][s] = make_pair(res_weight, w);
256  }
257  return;
258  }
